1""" 

2Django management command to create global superadmin. 

3 

4Usage: 

5 python manage.py create_superadmin 

6 

7Environment Variables: 

8 SUPERADMIN_USERNAME (default: 'superadmin') 

9 SUPERADMIN_EMAIL (default: 'admin@bookshelf.com') 

10 SUPERADMIN_PASSWORD (required) 

11 SUPERADMIN_FIRST_NAME (default: 'Super') 

12 SUPERADMIN_LAST_NAME (default: 'Admin') 

13 SUPERADMIN_PHONE (default: '0000000000') 

14 

15Features: 

16 - Idempotent (safe to run multiple times) 

17 - Environment-based credentials 

18 - Creates global admin (tenant=None) 

19""" 

20 

21import environ 

22from django.core.management.base import BaseCommand, CommandError 

23from django.contrib.auth import get_user_model

25User = get_user_model() 

26 

27# Initialize environ 

28env = environ.Env() 

29 

30 

31class Command(BaseCommand): 

32 

33 def handle(self, *args, **options): 

34 username = env.str('SUPERADMIN_USERNAME', default='superadmin') 

35 email = env.str('SUPERADMIN_EMAIL', default='admin@bookshelf.com') 

36 password = env.str('SUPERADMIN_PASSWORD', default=None) 

37 first_name = env.str('SUPERADMIN_FIRST_NAME', default='Super')

39 phone_no = env.str('SUPERADMIN_PHONE', default='0000000000') 

40 

41 if not password:

43 'SUPERADMIN_PASSWORD environment variable is required.' 

44 ) 

45 

46 if User.all_objects.filter(username=username).exists():

52 return

54 try: 

55 superadmin = User.objects.create_superuser( 

56 username=username, 

57 email=email, 

58 password=password, 

59 first_name=first_name, 

60 last_name=last_name, 

61 phone_no=phone_no, 

62 tenant=None 

63 ) 

64 

65 self.stdout.write( 

66 self.style.SUCCESS( 

67 f'Superadmin "{username}" created successfully' 

68 ) 

69 )
